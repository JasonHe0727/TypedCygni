def SumN(n: Int): Int
{
    var i = 1
    var sum = 0
    while (i <= n)
    {
        var k = 1
        sum = sum + i * k
        if (k == 1)
        {
            var m = 0
            sum = sum + m
        }
    }
    return sum
}
